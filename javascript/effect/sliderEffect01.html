    setInterval(()=>{
        // 1를 5로 나누면 몫은 0이고 나머지는 1
        // 2나누기5
        // 2를 5로 나누어도 몫은 0 나머지는 2
        //나눌수가 없어서 그대로 나머지가 되는 것입니다. 5이상의 수일때만 나눌 수가 있기 때문에 1부터 4까지는 5보다 작아서 나눌 수가 없으므로 그대로 나머지가 됩니다.
        //근데 5는 1~4를 나눌수가 없으니까 그대로 1~4가 나머지가 되는것이고

        let nextIndex = (currentIndex + 1) % sliderCount;
        slider[currentIndex].style.opacity = "0";
        slider[nextIndex].style.opacity = "1";


        currentIndex=nextIndex;
        
    },1000) // 1=(1000)초에 한번씩 실행
