{
    const num = [1, 2, 3, 4];

    function solution(arr, divisor) {
        var answer = [];

        if(num.length<4){
            for(i=0; i<arr.length; i++){
                if(arr[i] % divisor ==0) answer.push(arr[i]);
            }
        } else{
            for(i=0; i<arr.length; i++){
                if(arr[i] % divisor ==1) answer.unshift(arr[i]);
            }
        }

        return answer.shift();
    }

}
