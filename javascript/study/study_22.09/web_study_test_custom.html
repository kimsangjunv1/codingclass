03. 오늘 파티 참가 가능한 인원수는?

{
    function party() {
        let join = 0;
        let people = ['철수', '민수', '상수', '변수', '무리수']
        let enable = 4;

        if(people.length<=5){
            enable = 3;
        } else{
            enable = 8;
        }
        while (join < people.length) {
            join ++;
            if (join == enable) break;
        }
        console.log("오늘 파티 참가 가능한 인원수는 총 "+join+"명");
    }
    party();     
}
