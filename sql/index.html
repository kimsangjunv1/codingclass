MySQL

MySQL은 데이터 소프트웨어입니다. 일반적으로 데이터를 추가하거나 검색, 추출하는 기능을 모두 포함해서 데이터베이스라고 부릅니다.

MySQL은 세계에서 가장 많이 쓰이는 오픈 소스의 관계형 데이터베이스 관리시스템(RDBMS)입니다. MySQL은 PHP 스크립트 언어와 상호 연동이 잘 되면서 오픈소스를 개발된 무료 프로그램입니다. 그래서 홈페이지나 쇼핑몰(워드프레스, cafe24, 제로보드, 그누보드)등 일반적으로 웹 개발에 널리 사용하고 있습니다.

이름 중에 '김'으로 시작하는 멤버 출력

SELECT * FROM member WHERE youName LIKE '김%';

이름 중에 '김'씨 이거나 '황'씨인 경우 멤버 출력

SELECT * FROM member WHERE youName LIKE '김%' or youName LIKE '황%';

이름 중간에 '상'씨인 경우 멤버 출력

SELECT * FROM member WHERE youName LIKE '_상_'; 
    SELECT * FROM member WHERE youName LIKE '__연';

이름 중간에 데이터가 없는 경우 출력

SELECT * FROM member WHERE IS NULL;
SELECT * FROM member WHERE youName IS NOT NULL;

멤버 중에 나이가 30 이상이면서 김씨인 멤버 출력

SELECT * FROM member WHERE IS NULL;
SELECT * FROM member WHERE youName IS NOT NULL;

멤버 중에 나이가 18 이상이면서 김씨인 멤버 출력

SELECT * FROM member WHERE youAge >=18 AND youName LIKE '김%';

멤버 중에 나이가 18 이상이거나 김씨인 멤버 출력

SELECT * FROM member WHERE youAge >=18 OR youName LIKE '김%';

멤버 중에 나이가 20이상 이면서 (아이다가 2 또는 10인 멤버)

SELECT * FROM member WHERE (myMemberID = 2 OR myMemberID = 10) AND youAge >= 20;

SELECT * FROM member LIMIT 5, 5;

테이블 수정하기

UPDATE 테이블명(필드명) SET 컬럼명 조건

모든 회원의 핸드폰 번호 변경

UPDATE member SET youPhone = "01000000000";

아이디가 3번인 회원의 비밀번호를 '0123'

UPDATE member SET youPass = "0123" WHERE myMemberID = 3;

아이디가 3번인 회원의 비밀번호를 '1234567'로 변경

UPDATE member SET youPass = "1234567" WHERE myMemberID between 5 and 10;;

비밀번호를 "1234567"로 변경하고 내림차순으로 10개만 변경하기

UPDATE member SET youPass = "1234567" ORDER BY youPass DESC LIMIT 10;

테이블 삭제하기

DELETE FROM 테이블명 조건

회원 아이디가 5번인 멤버 삭제

delete from member where myMemberID between 7 and 9;

회원 비밀번호가 '1234'인 멤버 삭제

delete from member where youPass='1234';

회원 이름을 기준으로 5번째까지 멤버 삭제

delete from member ORDER BY youName LIMIT 5;

지정한 멤버 삭제

delete from member WHERE myMemberID IN(7,9,10);

테이블 수정

필드 추가하기

ALTER TABLE 테이블명 ADD 추가할 필드명 AFTER 필드 위치

member 테이블 나이 위에 핸드폰 번호 필드를 추가

ALTER TABLE member ADD youPhone int(20) NOT NULL AFTER youAge;

member 테이블 핸드폰 번호 뒤에 성별 필드를 추가

ALTER TABLE member ADD youGender enum('m','w','x') default 'x' comment '남자 m, 여자 w' AFTER youPhone;

필드 수정하가

ALTER TABLE 테이블명 MODIFY 수정할 필드명

전체 데이터 불러오기

ALTER TABLE member MODIFY youAge int(10);

필드 삭제하기

ALTER TABLE 테이블명 DROP 삭제시킬 필드명;

전체 데이터 불러오기

ALTER TABLE member DROP youGender;
